from PyQt5.QtWidgets import QMainWindow, QApplication, QLabel, QVBoxLayout, QWidget
from sys import argv
from callermainwindow import Ui_MainWindow


class MainWindow(QMainWindow, Ui_MainWindow):
    def __init__(self):
        super(MainWindow, self).__init__()
        self.setupUi(self)
        self.setWindowTitle("Window 1")

        # Set new layout. Do not use MainWindow layout
        layout = QVBoxLayout()
a
        # Item to be added either button, textEdit, pushButton
        label_1 = QLabel("Hello World")
        layout.addWidget(label_1)

        # Set new widget to MainWindow and add the new layout
        widget = QWidget()
        widget.setLayout(layout)

        # Set that new widget to centralwidget of MainWindow    
        self.setCentralWidget(widget)


def main():
    app = QApplication(argv)

    window_a = MainWindow()
    window_a.show()

    app.exec_()


if __name__ == "__main__":
    main()
